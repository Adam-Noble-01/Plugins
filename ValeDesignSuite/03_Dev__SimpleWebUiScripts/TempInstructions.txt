Util__MakeScaleAccuratePaperTemplates.html

Core Features:
- Browser Based UI allowing for selection of paper size.
- Uses ISO A Series Paper Sizes.
 - ISO A Series Paper Sizes are: A0, A1, A2, A3, A4, A5  
   - (So A0- A5 are the only sizes available)
 - ISO A Series Paper Sizes are in millimeters.
 - Allows for manual override of the paper size using a text input field for width and height in mm.
   - If nothing entered then the boxes dynamically report the paper size in mm as selected by the A series paper size selector.
- Dimensions are reported in mm as the paper is adjusted with clear dimensions lines similar to an architect's drawing.
- Centred on the page is a clear title reporting the paper size "A1 Landscape"and subtitle reporting the paper size in mm.
- Features a Landscape and Portrait mode selector.
- Has a Download button that downloads the paper template as a PNG Image.
- Has a Download button that downloads the paper template as a PDF.

            <script>
                // -----------------------------------------------------------------------------
                // REGION | Module Constants & Configuration
                // -----------------------------------------------------------------------------

                    // MODULE CONSTANTS | ISO A-Series Paper Sizes (mm)
                    // ------------------------------------------------------------
                    const PAPER_SIZES_MM = {
                        "A0": { width: 841, height: 1189 },
                        "A1": { width: 594, height: 841 },
                        "A2": { width: 420, height: 594 },
                        "A3": { width: 297, height: 420 },
                        "A4": { width: 210, height: 297 },
                        "A5": { width: 148, height: 210 }
                    };
                    // endregion ----------------------------------------------------

                // -----------------------------------------------------------------------------
                // REGION | Module Variables
                // -----------------------------------------------------------------------------

                    // MODULE VARIABLES | State Tracking Variables
                    // ------------------------------------------------------------
                    let currentSizeKey      = 'A4';                       // <-- Default paper size key
                    let currentOrientation  = 'portrait';                 // <-- Default orientation
                    // endregion ----------------------------------------------------

                // -----------------------------------------------------------------------------
                // REGION | Helper Functions
                // -----------------------------------------------------------------------------

                    // HELPER FUNCTION | Get Size Object Considering Orientation
                    // ------------------------------------------------------------
                    function getSizeMm(sizeKey, orientation) {
                        const baseSize = PAPER_SIZES_MM[sizeKey];
                        if (!baseSize) return { width: 210, height: 297 };  // Fallback to A4
                        return orientation === 'landscape'
                            ? { width: baseSize.height, height: baseSize.width }
                            : { ...baseSize };
                    }
                    // ---------------------------------------------------------------

                    // HELPER FUNCTION | Update Size Select Dropdown Options
                    // ------------------------------------------------------------
                    function populateSizeSelect() {
                        const sizeSelect = document.getElementById('size-select');
                        sizeSelect.innerHTML = '';
                        Object.keys(PAPER_SIZES_MM).forEach(key => {
                            const option = document.createElement('option');
                            option.value = key;
                            option.textContent = key;
                            if (key === currentSizeKey) option.selected = true;
                            sizeSelect.appendChild(option);
                        });
                    }
                    // ---------------------------------------------------------------

                // -----------------------------------------------------------------------------
                // REGION | UI Update Functions
                // -----------------------------------------------------------------------------

                    // FUNCTION | Apply Paper Settings & Render Template
                    // ------------------------------------------------------------
                    function applyPaperSettings() {
                        // Get user selections / overrides
                        const sizeKeySelect = document.getElementById('size-select').value;
                        const orientationSelect = document.getElementById('orientation-select').value;
                        const customWidthInput = parseFloat(document.getElementById('custom-width').value) || null;
                        const customHeightInput = parseFloat(document.getElementById('custom-height').value) || null;

                        currentSizeKey = sizeKeySelect;
                        currentOrientation = orientationSelect;

                        let sizeMm = getSizeMm(currentSizeKey, currentOrientation);
                        if (customWidthInput && customHeightInput) {
                            sizeMm = { width: customWidthInput, height: customHeightInput };
                        }

                        renderPaper(sizeMm);
                    }
                    // ---------------------------------------------------------------

                    // SUB FUNCTION | Render Paper & Dimension Elements
                    // ---------------------------------------------------------------
                    function renderPaper({ width, height }) {
                        const paperContainer = document.getElementById('paper-container');
                        paperContainer.style.width  = `${width}mm`;
                        paperContainer.style.height = `${height}mm`;

                        // Update title & subtitle
                        const titleEl = document.getElementById('paper-title');
                        const subtitleEl = document.getElementById('paper-subtitle');
                        const orientLabel = currentOrientation.charAt(0).toUpperCase() + currentOrientation.slice(1);
                        const titleText = customTitle(width, height) || `${currentSizeKey} ${orientLabel}`;
                        titleEl.textContent = titleText;
                        subtitleEl.textContent = `${width} mm × ${height} mm`;

                        // Update dimension lines positions
                        updateDimensionLines(width, height);
                        updateDimensionText(width, height);
                    }
                    // ---------------------------------------------------------------

                    // HELPER FUNCTION | Determine Custom Title (if overrides)
                    // ------------------------------------------------------------
                    function customTitle(width, height) {
                        const base = PAPER_SIZES_MM[currentSizeKey];
                        if (base && ((currentOrientation === 'portrait' && base.width === width && base.height === height) ||
                                       (currentOrientation === 'landscape' && base.width === height && base.height === width))) {
                            return null;                                    // <-- Using standard size, default title fine
                        }
                        return `Custom ${width}×${height}`;                 // <-- Custom size title
                    }
                    // ---------------------------------------------------------------

                    // SUB FUNCTION | Update Dimension Lines Positions
                    // ---------------------------------------------------------------
                    function updateDimensionLines(width, height) {
                        const inset = 20;                                   // <-- 20 mm inset for dimension lines

                        const lineLeft   = document.getElementById('dim-line-left');
                        const lineRight  = document.getElementById('dim-line-right');
                        const lineTop    = document.getElementById('dim-line-top');
                        const lineBottom = document.getElementById('dim-line-bottom');

                        lineLeft.style.left   = `${inset}mm`;
                        lineLeft.style.top    = `${inset}mm`;
                        lineLeft.style.height = `calc(${height}mm - ${inset * 2}mm)`;

                        lineRight.style.right  = `${inset}mm`;
                        lineRight.style.top    = `${inset}mm`;
                        lineRight.style.height = `calc(${height}mm - ${inset * 2}mm)`;

                        lineTop.style.left    = `${inset}mm`;
                        lineTop.style.top     = `${inset}mm`;
                        lineTop.style.width   = `calc(${width}mm - ${inset * 2}mm)`;

                        lineBottom.style.left   = `${inset}mm`;
                        lineBottom.style.bottom = `${inset}mm`;
                        lineBottom.style.width  = `calc(${width}mm - ${inset * 2}mm)`;
                    }
                    // ---------------------------------------------------------------

                    // SUB FUNCTION | Update Dimension Text Values & Positions
                    // ---------------------------------------------------------------
                    function updateDimensionText(width, height) {
                        const inset = 50;
                        const textVert = document.getElementById('dim-text-vertical');
                        const textHor  = document.getElementById('dim-text-horizontal');

                        // Vertical text (Height)
                        textVert.textContent = `${height} mm`;
                        textVert.style.left  = `${inset - 5}mm`;             // <-- Slightly offset for readability
                        textVert.style.top   = `50%`;

                        // Horizontal text (Width)
                        textHor.textContent = `${width} mm`;
                        textHor.style.top   = `${inset - 5}mm`;
                        textHor.style.left  = `50%`;
                    }
                    // ---------------------------------------------------------------

                // -----------------------------------------------------------------------------
                // REGION | Download Functions
                // -----------------------------------------------------------------------------

                    // FUNCTION | Download Paper as PNG
                    // ------------------------------------------------------------
                    function downloadPNG() {
                        const paperContainer = document.getElementById('paper-container');
                        html2canvas(paperContainer, { scale: 2 }).then(canvas => {
                            const link = document.createElement('a');
                            link.download = `${currentSizeKey}_${currentOrientation}.png`;
                            link.href = canvas.toDataURL();
                            link.click();
                        });
                    }
                    // ---------------------------------------------------------------

                    // FUNCTION | Download Paper as PDF
                    // ------------------------------------------------------------
                    function downloadPDF() {
                        const paperContainer = document.getElementById('paper-container');
                        html2canvas(paperContainer, { scale: 2 }).then(canvas => {
                            const imgData = canvas.toDataURL('image/png');
                            const { jsPDF } = window.jspdf;
                            const pdf = new jsPDF({ orientation: currentOrientation });
                            const widthMm = pdf.internal.pageSize.getWidth();
                            const heightMm = pdf.internal.pageSize.getHeight();
                            pdf.addImage(imgData, 'PNG', 0, 0, widthMm, heightMm);
                            pdf.save(`${currentSizeKey}_${currentOrientation}.pdf`);
                        });
                    }
                    // ---------------------------------------------------------------

                // -----------------------------------------------------------------------------
                // REGION | Event Binding & Initialization
                // -----------------------------------------------------------------------------

                    // FUNCTION | Initialize Application
                    // ------------------------------------------------------------
                    function init() {
                        populateSizeSelect();
                        applyPaperSettings();

                        document.getElementById('apply-button').addEventListener('click', applyPaperSettings);
                        document.getElementById('download-png').addEventListener('click', downloadPNG);
                        document.getElementById('download-pdf').addEventListener('click', downloadPDF);
                    }
                    // ---------------------------------------------------------------

                    // Immediately Initialize on DOMContentLoaded
                    document.addEventListener('DOMContentLoaded', init);

            </script>

                <div id="dim-text-horizontal-bottom"class="dimension-text horizontal red">210 mm</div>
            </div>

            <script>
                // -----------------------------------------------------------------------------
                // REGION | Module Constants & Configuration
                // -----------------------------------------------------------------------------

                    // MODULE CONSTANTS | ISO A-Series Paper Sizes (mm)
                    // ------------------------------------------------------------
                    const PAPER_SIZES_MM = {
                        "A0": { width: 841, height: 1189 },
                        "A1": { width: 594, height: 841 },
                        "A2": { width: 420, height: 594 },
                        "A3": { width: 297, height: 420 },
                        "A4": { width: 210, height: 297 },
                        "A5": { width: 148, height: 210 }
                    };
                    // endregion ----------------------------------------------------

                    // MODULE CONSTANTS | Output Quality Settings
                    // ------------------------------------------------------------
                    const OUTPUT_DPI = 300;                                               // <-- Print quality DPI for PNG output
                    const MM_TO_PIXELS_FACTOR = OUTPUT_DPI / 25.4;                      // <-- Convert mm to pixels at 300 DPI
                    // endregion ----------------------------------------------------

                // -----------------------------------------------------------------------------
                // REGION | Module Variables
                // -----------------------------------------------------------------------------

                    // MODULE VARIABLES | State Tracking Variables
                    // ------------------------------------------------------------
                    let currentSizeKey      = 'A4';                       // <-- Default paper size key
                    let currentOrientation  = 'portrait';                 // <-- Default orientation
                    // endregion ----------------------------------------------------

                // -----------------------------------------------------------------------------
                // REGION | Helper Functions
                // -----------------------------------------------------------------------------

                    // HELPER FUNCTION | Get Size Object Considering Orientation
                    // ------------------------------------------------------------
                    function getSizeMm(sizeKey, orientation) {
                        const baseSize = PAPER_SIZES_MM[sizeKey];
                        if (!baseSize) return { width: 210, height: 297 };  // Fallback to A4
                        return orientation === 'landscape'
                            ? { width: baseSize.height, height: baseSize.width }
                            : { ...baseSize };
                    }
                    // ---------------------------------------------------------------

                    // HELPER FUNCTION | Update Size Select Dropdown Options
                    // ------------------------------------------------------------
                    function populateSizeSelect() {
                        const sizeSelect = document.getElementById('size-select');
                        sizeSelect.innerHTML = '';
                        Object.keys(PAPER_SIZES_MM).forEach(key => {
                            const option = document.createElement('option');
                            option.value = key;
                            option.textContent = key;
                            if (key === currentSizeKey) option.selected = true;
                            sizeSelect.appendChild(option);
                        });
                    }
                    // ---------------------------------------------------------------

                // -----------------------------------------------------------------------------
                // REGION | UI Update Functions
                // -----------------------------------------------------------------------------

                    // FUNCTION | Apply Paper Settings & Render Template
                    // ------------------------------------------------------------
                    function applyPaperSettings() {
                        // Get user selections / overrides
                        const sizeKeySelect = document.getElementById('size-select').value;
                        const orientationSelect = document.getElementById('orientation-select').value;
                        const customWidthInput = parseFloat(document.getElementById('custom-width').value) || null;
                        const customHeightInput = parseFloat(document.getElementById('custom-height').value) || null;

                        currentSizeKey = sizeKeySelect;
                        currentOrientation = orientationSelect;

                        let sizeMm = getSizeMm(currentSizeKey, currentOrientation);
                        if (customWidthInput && customHeightInput) {
                            sizeMm = { width: customWidthInput, height: customHeightInput };
                        }

                        renderPaper(sizeMm);
                    }
                    // ---------------------------------------------------------------

                    // SUB FUNCTION | Render Paper & Dimension Elements
                    // ---------------------------------------------------------------
                    function renderPaper({ width, height }) {
                        const paperContainer = document.getElementById('paper-container');
                        paperContainer.style.width  = `${width}mm`;
                        paperContainer.style.height = `${height}mm`;

                        // Update title & subtitle
                        const titleEl = document.getElementById('paper-title');
                        const subtitleEl = document.getElementById('paper-subtitle');
                        const orientLabel = currentOrientation.charAt(0).toUpperCase() + currentOrientation.slice(1);
                        const titleText = customTitle(width, height) || `${currentSizeKey} ${orientLabel}`;
                        titleEl.textContent = titleText;
                        subtitleEl.textContent = `${width} mm × ${height} mm`;

                        // Update dimension lines positions
                        updateDimensionLines(width, height);
                        updateDimensionText(width, height);
                    }
                    // ---------------------------------------------------------------

                    // HELPER FUNCTION | Determine Custom Title (if overrides)
                    // ------------------------------------------------------------
                    function customTitle(width, height) {
                        const base = PAPER_SIZES_MM[currentSizeKey];
                        if (base && ((currentOrientation === 'portrait' && base.width === width && base.height === height) ||
                                       (currentOrientation === 'landscape' && base.width === height && base.height === width))) {
                            return null;                                    // <-- Using standard size, default title fine
                        }
                        return `Custom ${width}×${height}`;                 // <-- Custom size title
                    }
                    // ---------------------------------------------------------------

                    // SUB FUNCTION | Update Dimension Lines Positions
                    // ---------------------------------------------------------------
                    function updateDimensionLines(width, height) {
                        const inset = 20;                                   // <-- 20 mm inset for dimension lines

                        const lineLeft   = document.getElementById('dim-line-left');
                        const lineRight  = document.getElementById('dim-line-right');
                        const lineTop    = document.getElementById('dim-line-top');
                        const lineBottom = document.getElementById('dim-line-bottom');

                        // Vertical Lines
                        lineLeft.style.left   = `${inset}mm`;
                        lineLeft.style.top    = `0`;
                        lineLeft.style.height = `${height}mm`;             // Extend full height

                        lineRight.style.left   = `calc(${width}mm - ${inset}mm)`;
                        lineRight.style.top    = `0`;
                        lineRight.style.height = `${height}mm`;

                        // Horizontal Lines
                        lineTop.style.left    = `0`;
                        lineTop.style.top     = `${inset}mm`;
                        lineTop.style.width   = `${width}mm`;              // Extend full width

                        lineBottom.style.left   = `0`;
                        lineBottom.style.top    = `calc(${height}mm - ${inset}mm)`;
                        lineBottom.style.width  = `${width}mm`;
                    }
                    // ---------------------------------------------------------------

                    // SUB FUNCTION | Update Dimension Text Values & Positions
                    // ---------------------------------------------------------------
                    function updateDimensionText(width, height) {
                        const inset = 20;                                   // <-- 20 mm inset for dimension lines
                        const offset = 5;                                   // <-- 5 mm text offset from lines

                        const textVertLeft   = document.getElementById('dim-text-vertical-left');
                        const textVertRight  = document.getElementById('dim-text-vertical-right');
                        const textHorTop     = document.getElementById('dim-text-horizontal-top');
                        const textHorBottom  = document.getElementById('dim-text-horizontal-bottom');

                        // Vertical text – Left (green)
                        textVertLeft.textContent = `${height} mm`;
                        textVertLeft.style.left  = `${inset - offset}mm`;  // Offset inside
                        textVertLeft.style.top   = `50%`;

                        // Vertical text – Right (green)
                        textVertRight.textContent = `${height} mm`;
                        textVertRight.style.left  = `calc(${width}mm - ${inset - offset}mm)`;
                        textVertRight.style.top   = `50%`;

                        // Horizontal text – Top (red)
                        textHorTop.textContent = `${width} mm`;
                        textHorTop.style.top   = `${inset - offset}mm`;   // Offset above line
                        textHorTop.style.left  = `50%`;

                        // Horizontal text – Bottom (red)
                        textHorBottom.textContent = `${width} mm`;
                        textHorBottom.style.top   = `calc(${height}mm - ${inset - offset}mm)`; // Offset inside
                        textHorBottom.style.left  = `50%`;
                    }
                    // ---------------------------------------------------------------

                // -----------------------------------------------------------------------------
                // REGION | Download Functions
                // -----------------------------------------------------------------------------

                    // HELPER FUNCTION | Convert Millimeters to Pixels at Output DPI
                    // ------------------------------------------------------------
                    function mmToPixels(mm) {
                        return Math.round(mm * MM_TO_PIXELS_FACTOR);      // <-- Convert mm to pixels at 300 DPI
                    }
                    // ---------------------------------------------------------------

                    // FUNCTION | Download Paper as PNG
                    // ------------------------------------------------------------
                    function downloadPNG() {
                        const paperContainer = document.getElementById('paper-container');
                        
                        console.log('Starting PNG download...');                          // <-- Debug log
                        console.log('html2canvas available:', typeof html2canvas);       // <-- Check library
                        
                        // Check if html2canvas is available
                        if (typeof html2canvas === 'undefined') {
                            alert('html2canvas library not loaded. Please refresh the page and try again.');
                            return;
                        }
                        
                        // Use high-resolution canvas rendering without changing container
                        html2canvas(paperContainer, { 
                            scale: 2,                                                     // <-- Reduced scale for better compatibility
                            useCORS: true,
                            allowTaint: true,                                             // <-- Allow taint for local files
                            backgroundColor: '#ffffff',
                            logging: true,                                                // <-- Enable html2canvas logging
                            width: paperContainer.offsetWidth,                           // <-- Explicit dimensions
                            height: paperContainer.offsetHeight
                        }).then(canvas => {
                            console.log('Canvas generated successfully');                 // <-- Debug log
                            console.log('Canvas dimensions:', canvas.width, 'x', canvas.height);
                            
                            try {
                                // Create download link
                                const link = document.createElement('a');
                                const timestamp = new Date().toISOString().slice(0,10);      // <-- Add date stamp
                                link.download = `ValeTemplate_${currentSizeKey}_${currentOrientation}_${timestamp}.png`;
                                
                                // Convert canvas to blob for better browser compatibility
                                canvas.toBlob(function(blob) {
                                    if (blob) {
                                        const url = URL.createObjectURL(blob);
                                        link.href = url;
                                        
                                        // Trigger download
                                        document.body.appendChild(link);                     // <-- Add to DOM temporarily
                                        link.click();
                                        document.body.removeChild(link);                     // <-- Clean up
                                        
                                        // Clean up blob URL
                                        setTimeout(() => URL.revokeObjectURL(url), 100);
                                        
                                        console.log('PNG download initiated successfully');  // <-- Debug log
                                    } else {
                                        throw new Error('Failed to create blob from canvas');
                                    }
                                }, 'image/png', 0.95);
                                
                            } catch (error) {
                                console.error('Download creation failed:', error);           // <-- Error log
                                alert(`Download creation failed: ${error.message}`);        // <-- User notification
                            }
                        }).catch(error => {
                            console.error('PNG generation failed:', error);             // <-- Error log
                            console.error('Error details:', error.stack);               // <-- Full error stack
                            alert(`PNG download failed: ${error.message}\n\nCheck browser console for details.`);
                        });
                    }
                    // ---------------------------------------------------------------

                    // FUNCTION | Download Paper as PDF
                    // ------------------------------------------------------------
                    function downloadPDF() {
                        const paperContainer = document.getElementById('paper-container');
                        
                        console.log('Starting PDF download...');                         // <-- Debug log
                        
                        html2canvas(paperContainer, { 
                            scale: 2,                                                     // <-- Good resolution for PDF
                            useCORS: true,
                            allowTaint: false,
                            backgroundColor: '#ffffff'
                        }).then(canvas => {
                            console.log('Canvas generated for PDF');                     // <-- Debug log
                            
                            try {
                                const imgData = canvas.toDataURL('image/png');
                                const { jsPDF } = window.jspdf;
                                
                                // Get current paper dimensions for proper PDF sizing
                                const customWidth = parseFloat(document.getElementById('custom-width').value) || null;
                                const customHeight = parseFloat(document.getElementById('custom-height').value) || null;
                                let sizeMm = getSizeMm(currentSizeKey, currentOrientation);
                                
                                if (customWidth && customHeight) {
                                    sizeMm = { width: customWidth, height: customHeight };  // <-- Use custom dimensions
                                }
                                
                                // Create PDF with exact paper dimensions
                                const pdf = new jsPDF({
                                    orientation: currentOrientation,
                                    unit: 'mm',
                                    format: [sizeMm.width, sizeMm.height]                 // <-- Exact paper size
                                });
                                
                                // Add image to fill entire PDF page
                                pdf.addImage(imgData, 'PNG', 0, 0, sizeMm.width, sizeMm.height);
                                
                                const timestamp = new Date().toISOString().slice(0,10);  // <-- Add date stamp
                                pdf.save(`ValeTemplate_${currentSizeKey}_${currentOrientation}_${timestamp}.pdf`);
                                
                                console.log('PDF download completed');                   // <-- Debug log
                            } catch (error) {
                                console.error('PDF generation failed:', error);         // <-- Error log
                                alert(`PDF download failed: ${error.message}`);         // <-- User notification
                            }
                        }).catch(error => {
                            console.error('Canvas generation for PDF failed:', error);  // <-- Error log
                            alert(`PDF download failed: ${error.message}`);             // <-- User notification
                        });
                    }

                        console.log('Simple download test completed');
                    }

                    // FUNCTION | Check Library Dependencies
                    // ------------------------------------------------------------
                    function checkLibraries() {
                        console.log('=== LIBRARY DEPENDENCY CHECK ===');
                        console.log('html2canvas available:', typeof html2canvas !== 'undefined');
                        console.log('jsPDF available:', typeof window.jspdf !== 'undefined');
                        
                        if (typeof html2canvas === 'undefined') {
                            console.error('❌ html2canvas library failed to load');
                            return false;
                        }
                        
                        if (typeof window.jspdf === 'undefined') {
                            console.error('❌ jsPDF library failed to load');
                            return false;
                        }
                        
                        console.log('✅ All libraries loaded successfully');
                        return true;
                    }
                    // ---------------------------------------------------------------

                    // FUNCTION | Initialize Application
                    // ------------------------------------------------------------
                    function init() {
                        console.log('🚀 Initializing Vale Paper Template Generator...');
                        
                        // Check if required libraries are loaded
                        const librariesOK = checkLibraries();
                        if (!librariesOK) {
                            console.warn('⚠️ Some libraries missing - download features may not work');
                        }
                        
                        populateSizeSelect();
                        applyPaperSettings();

                        // Bind event listeners
                        document.getElementById('apply-button').addEventListener('click', applyPaperSettings);
                        document.getElementById('download-png').addEventListener('click', downloadPNG);
                        document.getElementById('download-pdf').addEventListener('click', downloadPDF);
                        
                        // Add test download for debugging (optional - remove in production)
                        console.log('💡 To test download functionality, run: testSimpleDownload()');
                        
                        console.log('✅ Initialization complete');
                    }
                    // ---------------------------------------------------------------

                    // Immediately Initialize on DOMContentLoaded
                    document.addEventListener('DOMContentLoaded', init);

            </script>
